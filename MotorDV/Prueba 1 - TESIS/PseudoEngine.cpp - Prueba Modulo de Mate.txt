    std::cout << "===============================\n";

    std::cout << "----- Prueba de MathUtils -----\n\n";

    float valor = 25.0f;

    std::cout << "[sqrt_approx]\n";
    std::cout << "Valor usado: " << valor << "\n";
    std::cout << "Resultado aproximado: " << sqrt_approx(valor) << "\n\n";

    std::cout << "[inv_sqrt_approx]\n";
    std::cout << "Valor usado: " << valor << "\n";
    std::cout << "Resultado aproximado: " << inv_sqrt_approx(valor) << "\n\n";

    float angulo = 1.0f;

    std::cout << "[sin_approx]\n";
    std::cout << "Angulo en radianes: " << angulo << "\n";
    std::cout << "Resultado aproximado: " << sin_approx(angulo) << "\n\n";

    std::cout << "-------- Prueba de Vec2 -------\n\n";

    std::cout << "[Vectores usados]\n";
    std::cout << "Vector a = (" << a.x << ", " << a.y << ")\n";
    std::cout << "Vector b = (" << b.x << ", " << b.y << ")\n\n";

    std::cout << "[a + b]\nResultado: (" << suma.x << ", " << suma.y << ")\n\n";

    std::cout << "[a - b]\nResultado: (" << resta.x << ", " << resta.y << ")\n\n";

    float punto = a.dot(b);
    std::cout << "[Producto punto de a y b]\nResultado: " << punto << "\n\n";

    std::cout << "-------------------------------\n";

    return 0;
